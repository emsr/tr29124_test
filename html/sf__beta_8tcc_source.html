60  /**
61  * @brief Return the beta function: @f$ B(a,b) @f$.
62  *
63  * The beta function is defined by
64  * @f[
65  * B(a,b) = \int_0^1 t^{a - 1} (1 - t)^{b - 1} dt
66  * = \frac{\Gamma(a)\Gamma(b)}{\Gamma(a+b)}
67  * @f]
68  *
69  * @param __a The first argument of the beta function.
70  * @param __b The second argument of the beta function.
71  * @return The beta function.
72  */
73  template<typename _Tp>
74  _Tp
75  __beta_gamma(_Tp __a, _Tp __b)
76  {
77 
78  _Tp __bet;
79  if (__a > __b)
80  {
81  __bet = __gamma(__a) / __gamma(__a + __b);
82  __bet *= __gamma(__b);
83  }
84  else
85  {
86  __bet = __gamma(__b) / __gamma(__a + __b);
87  __bet *= __gamma(__a);
88  }
89 
90  return __bet;
91  }

94  * @brief Return the beta function @f$B(a,b)@f$ using
95  * the log gamma functions.
96  *
97  * The beta function is defined by
98  * @f[
99  * B(a,b) = \int_0^1 t^{a - 1} (1 - t)^{b - 1} dt
100  * = \frac{\Gamma(a)\Gamma(b)}{\Gamma(a+b)}
101  * @f]
102  *
103  * @param __a The first argument of the beta function.
104  * @param __b The second argument of the beta function.
105  * @return The beta function.

119  /**
120  * @brief Return the beta function @f$B(x,y)@f$ using
121  * the product form.
122  *
123  * The beta function is defined by
124  * @f[
125  * B(a,b) = \int_0^1 t^{a - 1} (1 - t)^{b - 1} dt
126  * = \frac{\Gamma(a)\Gamma(b)}{\Gamma(a+b)}
127  * @f]
128  * Here, we employ the product form:
129  * @f[
130  * B(a,b) = \frac{a + b}{a b} \prod_{k=1}^{\infty}
131  * \frac{1 + (a + b) / k}{(1 + a / k) (1 + b / k)}
132  * @f]
133  *
134  * @param __a The first argument of the beta function.
135  * @param __b The second argument of the beta function.
136  * @return The beta function.
137  */
138  template<typename _Tp>
139  _Tp
140  __beta_product(_Tp __a, _Tp __b)
141  {
142 
143  _Tp __bet = (__a + __b) / (__a * __b);
144 
145  const unsigned int _S_max_iter = 1000000;
146 
147  for (unsigned int __k = 1; __k < _S_max_iter; ++__k)
148  {
149  _Tp __term = (_Tp{1} + (__a + __b) / __k)
150  / ((_Tp{1} + __a / __k) * (_Tp{1} + __b / __k));
151  __bet *= __term;
152  }
153 
154  return __bet;
155  }

183  * Return the regularized incomplete beta function, @f$ I_x(a,b) @f$,
184  * of arguments @c a, @c b, and @c x.
185  *
186  *
187  * @param __a The first parameter
188  * @param __b The second parameter
189  * @param __x The argument
190  */
191  template<typename _Tp>
192  _Tp
193  __beta_inc_cont_frac(_Tp __a, _Tp __b, _Tp __x)
194  {
195  constexpr unsigned int _S_itmax = 100;
196  constexpr _Tp _S_eps = __gnu_cxx::__epsilon<_Tp>();
197 
198  _Tp __apb = __a + __b;
199  _Tp __ap1 = __a + _Tp{1};
200  _Tp __am1 = __a - _Tp{1};
201  _Tp __c = _Tp{1};
202  _Tp __d = _Tp{1} - __apb * __x / __ap1;
203  if (std::abs(__d) < _S_eps)
204  __d = _S_eps;
205  __d = _Tp{1} / __d;
206  _Tp __h = __d;
207  for (unsigned int __m = 1; __m <= _S_itmax; ++__m)
208  {
209  _Tp __m2(2 * __m);
210 
211  _Tp __aa = _Tp(__m) * (__b - _Tp(__m)) * __x
212  / ((__am1 + __m2) * (__a + __m2));
213  __d = _Tp{1} + __aa * __d;
214  if (std::abs(__d) < _S_eps)
215  __d = _S_eps;
216  __c = _Tp{1} + __aa / __c;
217  if (std::abs(__c) < _S_eps)
218  __c = _S_eps;
219  __d = _Tp{1} / __d;
220  __h *= __d * __c;
221 
222  __aa = -(__a + _Tp(__m)) * (__apb + __m) * __x
223  / ((__a + __m2) * (__ap1 + __m2));
224  __d = _Tp{1} + __aa * __d;
225  if (std::abs(__d) < _S_eps)
226  __d = _S_eps;
227  __c = _Tp{1} + __aa / __c;
228  if (std::abs(__c) < _S_eps)
229  __c = _S_eps;
230  __d = _Tp{1} / __d;
231  _Tp __del = __d * __c;
232  __h *= __del;
233 
234  if (std::abs(__del - _Tp{1}) < _S_eps)
235  return __h;
236  }
237  std::__throw_runtime_error(__N("__beta_inc_cont_frac: "
238  "continued fractions failed to converge"));

241  /**
242  * Return the regularized incomplete beta function, @f$ I_x(a,b) @f$,
243  * of arguments @c a, @c b, and @c x.
244  *
245  * The regularized incomplete beta function is defined by:
246  * @f[
247  * I_x(a,b) = \frac{B_x(a,b)}{B(a,b)}
248  * @f]
249  * where
250  * @f[
251  * B_x(a,b) = \int_0^x t^{a - 1} (1 - t)^{b - 1} dt
252  * @f]
253  * is the non-regularized beta function and @f$ B(a,b) @f$
254  * is the usual beta function.
255  *
256  * @param __a The first parameter
257  * @param __b The second parameter
258  * @param __x The argument
259  */
260  template<typename _Tp>
261  _Tp
262  __beta_inc(_Tp __a, _Tp __b, _Tp __x)
263  {
264  constexpr auto _S_NaN = __gnu_cxx::__quiet_NaN<_Tp>();
265 
266  if (__isnan(__x) || __isnan(__a) || __isnan(__b))
267  return _S_NaN;
268 
269  if (__x < _Tp{0} || __x > _Tp{1})
270  std::__throw_domain_error(__N("__beta_inc: argument out of range"));
271  else if (__x == _Tp{0} || __x == _Tp{1})
272  return _Tp{0};
273  else
274  {
275  auto __fact = std::exp(std::lgamma(__a + __b)
276  - std::lgamma(__a) - std::lgamma(__b)
277  + __a * std::log(__x) + __b * std::log(_Tp{1} - __x));
278 
279  if (__x < (__a + _Tp{1}) / (__a + __b + _Tp{2}))
280  return __fact * __beta_inc_cont_frac(__a, __b, __x) / __a;
281  else
282  return _Tp{1}
283  - __fact * __beta_inc_cont_frac(__b, __a, _Tp{1} - __x) / __b;
284  }
285  }
