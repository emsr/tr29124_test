46  * @brief This function computes the hyperbolic cosine @f$ Chi(x) @f$
47  * and hyperbolic sine @f$ Shi(x) @f$ integrals
48  * by continued fraction for positive argument.
49  */
50  ////FIXME!!!!
51  template<typename _Tp>
52  void
53  __chshint_cont_frac(_Tp __t, _Tp& _Chi, _Tp& _Shi)
54  {
55  constexpr unsigned int _S_max_iter = 100;
56  constexpr auto _S_eps = _Tp{5} * __gnu_cxx::__epsilon<_Tp>();
57  constexpr auto _S_fp_min = __gnu_cxx::__min<_Tp>();
58  constexpr auto _S_pi_2 = __gnu_cxx::__math_constants<_Tp>::__pi_half;
59 
60  // Evaluate Chi and Shi by Lentz's modified method of continued fracions.
61  std::complex<_Tp> __b(_Tp{1}, __t);
62  std::complex<_Tp> __c(_Tp{1} / _S_fp_min);
63  std::complex<_Tp> __d(_Tp{1} / __b);
64  std::complex<_Tp> __h(__d);
65  unsigned int __i = 2;
66  while (true)
67  {
68  _Tp __a = -(__i - 1) * (__i - 1);
69  __b += _Tp{2};
70  __d = _Tp{1} / (__a * __d + __b);
71  __c = __b + __a / __c;
72  auto __del = __c * __d;
73  __h *= __del;
74  if (std::abs(__del.real() - _Tp{1}) + std::abs(__del.imag()) < _S_eps)
75  break;
76  if (__i > _S_max_iter)
77  std::__throw_runtime_error(__N("__chshint_cont_frac: "
78  "Continued fraction evaluation failed"));
79  ++__i;
80  }
81  __h *= std::polar(_Tp{1}, -__t);
82  _Chi = -__h.real();
83  _Shi = _S_pi_2 + __h.imag();
84 
85  return;
86  }

89  /**
90  * @brief This function computes the hyperbolic cosine @f$ Chi(x) @f$
91  * and hyperbolic sine @f$ Shi(x) @f$ integrals
92  * by series summation for positive argument.
93  */
94  template<typename _Tp>
95  void
96  __chshint_series(_Tp __t, _Tp& _Chi, _Tp& _Shi)
97  {
98  constexpr auto _S_max_iter = 100;
99  constexpr auto _S_eps = _Tp{5} * __gnu_cxx::__epsilon<_Tp>();
100  constexpr auto _S_fp_min = __gnu_cxx::__min<_Tp>();
101  constexpr auto _S_gamma_e = __gnu_cxx::__math_constants<_Tp>::__gamma_e;
102 
103  // Evaluate Chi and Shi by series simultaneously.
104  _Tp _Csum(0), _Ssum(0);
105  if (__t * __t < _S_fp_min)
106  {
107  // Avoid underflow.
108  _Csum = _Tp{0};
109  _Ssum = __t;
110  }
111  else
112  {
113  // Evaluate Shi and Chi by series expansion.
114  _Tp __sum(0);
115  _Tp __fact(1);
116  auto __odd = true;
117  auto __k = 1;
118  while (true)
119  {
120  __fact *= __t / __k;
121  _Tp __term = __fact / __k;
122  __sum += __term;
123  _Tp __err = __term / std::abs(__sum);
124  if (__odd)
125  {
126  _Ssum = __sum;
127  __sum = _Csum;
128  }
129  else
130  {
131  _Csum = __sum;
132  __sum = _Ssum;
133  }
134  if (__err < _S_eps)
135  break;
136  __odd = !__odd;
137  ++__k;
138  if (__k > _S_max_iter)
139  std::__throw_runtime_error(__N("__chshint_series: "
140  "Series evaluation failed"));
141  }
142  }
143  _Chi = _S_gamma_e + std::log(__t) + _Csum;
144  _Shi = _Ssum;
145 
146  return;
147  }

150  /**
151  * @brief This function returns the hyperbolic cosine @f$ Ci(x) @f$
152  * and hyperbolic sine @f$ Si(x) @f$ integrals as a pair.
153  *
154  * The hyperbolic cosine integral is defined by:
155  * @f[
156  * Chi(x) = \gamma_E + \log(x) + \int_0^x dt \frac{\cosh(t) - 1}{t}
157  * @f]
158  *
159  * The hyperbolic sine integral is defined by:
160  * @f[
161  * Shi(x) = \int_0^x dt \frac{\sinh(t)}{t}
162  * @f]
163  */
164  template<typename _Tp>
165  std::pair<_Tp, _Tp>
166  __chshint(_Tp __x, _Tp& _Chi, _Tp& _Shi)
167  {
168  constexpr auto _S_NaN = __gnu_cxx::__quiet_NaN<_Tp>();
169  if (__isnan(__x))
170  return std::make_pair(_S_NaN, _S_NaN);
171 
172  auto __t = std::abs(__x);
173  if (__t == _Tp{0})
174  {
175  _Chi = -__gnu_cxx::__infinity<_Tp>();
176  _Shi = _Tp{0};
177  }
178  else if (__t > _Tp{2})
179  __chshint_cont_frac(__t, _Chi, _Shi);
180  else
181  __chshint_series(__t, _Chi, _Shi);
182 
183  if (__x < _Tp{0})
184  _Shi = -_Shi;
185 
186  return std::make_pair(_Chi, _Shi);
187  }
