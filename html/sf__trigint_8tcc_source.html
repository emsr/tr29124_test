51  /**
52  * @brief This function computes the sine @f$ Si(x) @f$
53  * and cosine @f$ Ci(x) @f$ integrals by continued fraction
54  * for positive argument.
55  */
56  template<typename _Tp>
57  void
58  __sincosint_cont_frac(_Tp __t, _Tp& _Si, _Tp& _Ci)
59  {
60  constexpr auto _S_max_iter = 100;
61  constexpr auto _S_eps = _Tp{5} * __gnu_cxx::__epsilon<_Tp>();
62  constexpr auto _S_fp_min = __gnu_cxx::__min<_Tp>();
63  constexpr auto _S_pi_2 = __gnu_cxx::__math_constants<_Tp>::__pi_half;
64 
65  // Evaluate Ci and Si by Lentz's modified method of continued fractions.
66  std::complex<_Tp> __b(_Tp{1}, __t);
67  std::complex<_Tp> __c(_Tp{1} / _S_fp_min);
68  std::complex<_Tp> __d(_Tp{1} / __b);
69  std::complex<_Tp> __h(__d);
70  int __i = 2;
71  while (true)
72  {
73  auto __a = -_Tp(__i - 1) * _Tp(__i - 1);
74  __b += _Tp{2};
75  __d = _Tp{1} / (__a * __d + __b);
76  __c = __b + __a / __c;
77  std::complex<_Tp> __del = __c * __d;
78  __h *= __del;
79  if (std::abs(__del - _Tp{1}) < _S_eps)
80  break;
81  if (__i > _S_max_iter)
82  std::__throw_runtime_error(__N("__sincosint_cont_frac: "
83  "continued fraction evaluation failed"));
84  ++__i;
85  }
86  __h *= std::polar(_Tp{1}, -__t);
87  _Ci = -__h.real();
88  _Si = _S_pi_2 + __h.imag();
89 
90  return;
91  }

94  /**
95  * @brief This function computes the sine @f$ Si(x) @f$
96  * and cosine @f$ Ci(x) @f$ integrals by series summation
97  * for positive argument.
98  */
99  template<typename _Tp>
100  void
101  __sincosint_series(_Tp __t, _Tp& _Si, _Tp& _Ci)
102  {
103  constexpr auto _S_max_iter = 100;
104  constexpr auto _S_eps = _Tp{5} * __gnu_cxx::__epsilon<_Tp>();
105  constexpr auto _S_fp_min = __gnu_cxx::__min<_Tp>();
106  constexpr auto _S_gamma_e
107  = __gnu_cxx::__math_constants<_Tp>::__gamma_e;
108 
109  // Evaluate Ci and Si by series simultaneously.
110  _Tp __sumc(0), __sums(0);
111  if (__t * __t < _S_fp_min)
112  {
113  // Avoid underflow.
114  __sumc = _Tp{0};
115  __sums = __t;
116  }
117  else
118  {
119  // Evaluate Si and Ci by series expansion.
120  _Tp __sum(0);
121  _Tp __sign(1), __fact(1);
122  bool __odd = true;
123  unsigned int __k = 1;
124  while (true)
125  {
126  __fact *= __t / __k;
127  _Tp __term = __fact / __k;
128  __sum += __sign * __term;
129  _Tp __err = __term / std::abs(__sum);
130  if (__odd)
131  {
132  __sign = -__sign;
133  __sums = __sum;
134  __sum = __sumc;
135  }
136  else
137  {
138  __sumc = __sum;
139  __sum = __sums;
140  }
141  if (__err < _S_eps)
142  break;
143  __odd = !__odd;
144  ++__k;
145  if (__k > _S_max_iter)
146  std::__throw_runtime_error(__N("__sincosint_series: "
147  "series evaluation failed"));
148  }
149  }
150  _Si = __sums;
151  _Ci = _S_gamma_e + std::log(__t) + __sumc;
152 
153  return;
154  }

157  /**
158  * @brief This function computes the sine @f$ Si(x) @f$
159  * and cosine @f$ Ci(x) @f$ integrals by asymptotic series summation
160  * for positive argument.
161  *
162  * The asymptotic series is very good for x > 50.
163  */
164  template<typename _Tp>
165  void
166  __sincosint_asymp(_Tp __t, _Tp& _Si, _Tp& _Ci)
167  {
168  constexpr auto _S_max_iter = 100;
169  constexpr auto _S_eps = _Tp{5} * __gnu_cxx::__epsilon<_Tp>();
170  constexpr auto _S_pi_2 = __gnu_cxx::__math_constants<_Tp>::__pi_half;
171 
172  auto __invt = _Tp{1} / __t;
173  auto __term = _Tp{1}; // 0!
174  auto __sume = _Tp{__term};
175  __term *= __invt; // 1! / t
176  auto __sumo = _Tp{__term};
177  auto __sign = _Tp{1};
178  auto __even = true;
179  auto __k = 2;
180  while (true)
181  {
182  __term *= __k * __invt;
183 
184  if (__even)
185  {
186  __sign = -__sign;
187  __sume += __sign * __term;
188 
189  }
190  else
191  {
192  __sumo += __sign * __term;
193  if (__term / std::abs(__sumo) < _S_eps)
194  break;
195  }
196 
197  __even = !__even;
198 
199  if (__k > _S_max_iter)
200  std::__throw_runtime_error(__N("__sincosint_asymp: "
201  "Series evaluation failed"));
202  ++__k;
203  }
204 
205  _Si = _S_pi_2
206  - std::cos(__t) * __invt * __sume
207  - std::sin(__t) * __invt * __sumo;
208  _Ci = std::sin(__t) * __invt * __sume
209  - std::cos(__t) * __invt * __sumo;
210 
211  return;
212  }

215  /**
216  * @brief This function returns the sine @f$ Si(x) @f$
217  * and cosine @f$ Ci(x) @f$ integrals as a @c pair.
218  *
219  * The sine integral is defined by:
220  * @f[
221  * Si(x) = \int_0^x dt \frac{\sin(t)}{t}
222  * @f]
223  *
224  * The cosine integral is defined by:
225  * @f[
226  * Ci(x) = \gamma_E + \log(x) + \int_0^x dt \frac{\cos(t) - 1}{t}
227  * @f]
228  */
229  template<typename _Tp>
230  std::pair<_Tp, _Tp>
231  __sincosint(_Tp __x)
232  {
233  constexpr auto _S_NaN = __gnu_cxx::__quiet_NaN<_Tp>();
234  if (__isnan(__x))
235  return std::make_pair(_S_NaN, _S_NaN);
236 
237  auto __t = std::abs(__x);
238  _Tp _Ci, _Si;
239  if (__t == _Tp{0})
240  {
241  _Si = _Tp{0};
242  _Ci = -__gnu_cxx::__infinity<_Tp>();
243  }
244  else if (__t > _Tp{1000}) // Check this!
245  __sincosint_asymp(__t, _Si, _Ci);
246  else if (__t > _Tp{2})
247  __sincosint_cont_frac(__t, _Si, _Ci);
248  else
249  __sincosint_series(__t, _Si, _Ci);
250 
251  if (__x < _Tp{0})
252  _Si = -_Si;
253 
254  return std::make_pair(_Si, _Ci);
255  }
