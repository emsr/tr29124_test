60  /**
61  * @brief Return the exponential integral @f$ E_1(x) @f$
62  * by series summation. This should be good
63  * for @f$ x < 1 @f$.
64  *
65  * The exponential integral is given by
66  * \f[
67  * E_1(x) = \int_{1}^{\infty} \frac{e^{-xt}}{t} dt
68  * \f]
69  *
70  * @param __x The argument of the exponential integral function.
71  * @return The exponential integral.
72  */
73  template<typename _Tp>
74  _Tp
75  __expint_E1_series(_Tp __x)
76  {
77  constexpr auto __eps = __gnu_cxx::__epsilon<_Tp>();
78  auto __term = _Tp{1};
79  auto __esum = _Tp{0};
80  auto __osum = _Tp{0};
81  const unsigned int __max_iter = 100;
82  for (unsigned int __i = 1; __i < __max_iter; ++__i)
83  {
84  __term *= - __x / __i;
85  if (std::abs(__term) < __eps)
86  break;
87  if (__term >= _Tp{0})
88  __esum += __term / __i;
89  else
90  __osum += __term / __i;
91  }
92 
93  return - __esum - __osum
94  - __gnu_cxx::__math_constants<_Tp>::__gamma_e - std::log(__x);
95  }

98  /**
99  * @brief Return the exponential integral @f$ E_1(x) @f$
100  * by asymptotic expansion.
101  *
102  * The exponential integral is given by
103  * \f[
104  * E_1(x) = \int_{1}^\infty \frac{e^{-xt}}{t} dt
105  * \f]
106  *
107  * @param __x The argument of the exponential integral function.
108  * @return The exponential integral.
109  */
110  template<typename _Tp>
111  _Tp
112  __expint_E1_asymp(_Tp __x)
113  {
114  auto __term = _Tp{1};
115  auto __esum = _Tp{1};
116  auto __osum = _Tp{0};
117  const unsigned int __max_iter = 1000;
118  for (unsigned int __i = 1; __i < __max_iter; ++__i)
119  {
120  auto __prev = __term;
121  __term *= - __i / __x;
122  if (std::abs(__term) > std::abs(__prev))
123  break;
124  if (__term >= _Tp{0})
125  __esum += __term;
126  else
127  __osum += __term;
128  }
129 
130  return std::exp(- __x) * (__esum + __osum) / __x;
131  }

134  /**
135  * @brief Return the exponential integral @f$ E_n(x) @f$
136  * by series summation.
137  *
138  * The exponential integral is given by
139  * \f[
140  * E_n(x) = \int_{1}^\infty \frac{e^{-xt}}{t^n} dt
141  * \f]
142  *
143  * @param __n The order of the exponential integral function.
144  * @param __x The argument of the exponential integral function.
145  * @return The exponential integral.
146  */
147  template<typename _Tp>
148  _Tp
149  __expint_En_series(unsigned int __n, _Tp __x)
150  {
151  const unsigned int __max_iter = 100;
152  constexpr auto __eps = __gnu_cxx::__epsilon<_Tp>();
153  const int __nm1 = __n - 1;
154  _Tp __ans = (__nm1 != 0
155  ? _Tp{1} / __nm1
156  : -std::log(__x) - __gnu_cxx::__math_constants<_Tp>::__gamma_e);
157  _Tp __fact = _Tp{1};
158  for (int __i = 1; __i <= __max_iter; ++__i)
159  {
160  __fact *= -__x / _Tp{__i};
161  _Tp __del;
162  if ( __i != __nm1 )
163  __del = -__fact / _Tp{__i - __nm1};
164  else
165  {
166  _Tp __psi = -__gnu_cxx::__math_constants<_Tp>::__gamma_e;
167  for (int __ii = 1; __ii <= __nm1; ++__ii)
168  __psi += _Tp{1} / _Tp(__ii);
169  __del = __fact * (__psi - std::log(__x));
170  }
171  __ans += __del;
172  if (std::abs(__del) < __eps * std::abs(__ans))
173  return __ans;
174  }
175  std::__throw_runtime_error(__N("__expint_En_series: "
176  "series summation failed"));
177  }

181  * @brief Return the exponential integral @f$ E_n(x) @f$
182  * by continued fractions.
183  *
184  * The exponential integral is given by
185  * \f[
186  * E_n(x) = \int_{1}^\infty \frac{e^{-xt}}{t^n} dt
187  * \f]
188  *
189  * @param __n The order of the exponential integral function.
190  * @param __x The argument of the exponential integral function.
191  * @return The exponential integral.
192  */
193  template<typename _Tp>
194  _Tp
195  __expint_En_cont_frac(unsigned int __n, _Tp __x)
196  {
197  const unsigned int __max_iter = 100;
198  constexpr auto __eps = __gnu_cxx::__epsilon<_Tp>();
199  constexpr auto __fp_min = __gnu_cxx::__min<_Tp>();
200  const int __nm1 = __n - 1;
201  auto __b = __x + _Tp(__n);
202  auto __c = _Tp{1} / __fp_min;
203  auto __d = _Tp{1} / __b;
204  auto __h = __d;
205  for ( unsigned int __i = 1; __i <= __max_iter; ++__i )
206  {
207  auto __a = -_Tp{__i * (__nm1 + __i)};
208  __b += _Tp{2};
209  __d = _Tp{1} / (__a * __d + __b);
210  __c = __b + __a / __c;
211  const auto __del = __c * __d;
212  __h *= __del;
213  if (std::abs(__del - _Tp{1}) < __eps)
214  {
215  const auto __ans = __h * std::exp(-__x);
216  return __ans;
217  }
218  }
219  std::__throw_runtime_error(__N("__expint_En_cont_frac: "
220  "continued fraction failed"));
221  }

224  /**
225  * @brief Return the exponential integral @f$ E_n(x) @f$
226  * by recursion. Use upward recursion for @f$ x < n @f$
227  * and downward recursion (Miller's algorithm) otherwise.
228  *
229  * The exponential integral is given by
230  * \f[
231  * E_n(x) = \int_{1}^\infty \frac{e^{-xt}}{t^n} dt
232  * \f]
233  *
234  * @param __n The order of the exponential integral function.
235  * @param __x The argument of the exponential integral function.
236  * @return The exponential integral.
237  */
238  template<typename _Tp>
239  _Tp
240  __expint_En_recursion(unsigned int __n, _Tp __x)
241  {
242  _Tp __En;
243  _Tp __E1 = __expint_E1(__x);
244  if (__x < _Tp(__n))
245  {
246  // Forward recursion is stable only for n < x.
247  __En = __E1;
248  for (unsigned int __j = 2; __j < __n; ++__j)
249  __En = (std::exp(-__x) - __x * __En) / _Tp(__j - 1);
250  }
251  else
252  {
253  // Backward recursion is stable only for n >= x.
254  __En = _Tp{1};
255  /// @todo Find a principled starting number
256  /// for the @f$ E_n(x) @f$ downward recursion.
257  const int __N = __n + 20;
258  _Tp __save = _Tp{0};
259  for (int __j = __N; __j > 0; --__j)
260  {
261  __En = (std::exp(-__x) - __j * __En) / __x;
262  if (__j == __n)
263  __save = __En;
264  }
265  _Tp __norm = __En / __E1;
266  __En /= __norm;
267  }
268 
269  return __En;
270  }
