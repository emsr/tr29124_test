65  /**
66  * A function to reliably compare two floating point numbers.
67  *
68  * @param __a The left hand side
69  * @param __b The right hand side
70  * @param __mul The multiplier for numeric epsilon for comparison
71  * @return @c true if a and b are equal to zero
72  * or differ only by @f$ max(a,b) * mul * epsilon @f$
73  */
74  template<typename _Tp>
75  inline bool
76  __fp_is_equal(_Tp __a, _Tp __b, _Tp __mul = _Tp{1})
77  {
78  const auto _S_tol = __mul * std::numeric_limits<_Tp>::epsilon();
79  bool __retval = true;
80  if ((__a != _Tp{0}) || (__b != _Tp{0}))
81  // Looks mean, but is necessary that the next line has sense.
82  __retval = (std::abs(__a - __b) < __fp_max_abs(__a, __b) * _S_tol);
83  return __retval;
84  }

86  /**
87  * A function to reliably compare a floating point number with zero.
88  *
89  * @param __a The floating point number
90  * @param __mul The multiplier for numeric epsilon for comparison
91  * @return @c true if a and b are equal to zero

93  */
94  template<typename _Tp>
95  inline bool
96  __fp_is_zero(_Tp __a, _Tp __mul = _Tp{1})
97  {
98  const auto _S_tol = __mul * std::numeric_limits<_Tp>::epsilon();
99  if (__a != _Tp{0})
