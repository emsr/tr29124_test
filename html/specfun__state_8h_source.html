377  /**
378  * @brief A structure for the gamma functions required by the Temme series
379  * expansions of @f$ N_\nu(x) @f$ and @f$ K_\nu(x) @f$.
380  * @f[
381  * \Gamma_1 = \frac{1}{2\mu}
382  * \left[\frac{1}{\Gamma(1 - \mu)} - \frac{1}{\Gamma(1 + \mu)}\right]
383  * @f]
384  * and
385  * @f[
386  * \Gamma_2 = \frac{1}{2}
387  * \left[\frac{1}{\Gamma(1 - \mu)} + \frac{1}{\Gamma(1 + \mu)}\right]
388  * @f]
389  * where @f$ -1/2 <= \mu <= 1/2 @f$ is @f$ \mu = \nu - N @f$ and @f$ N @f$.
390  * is the nearest integer to @f$ \nu @f$.
391  * The values of @f$ \Gamma(1 + \mu) @f$ and @f$ \Gamma(1 - \mu) @f$
392  * are returned as well.
393  *
394  * The accuracy requirements on this are high for @f$ |\mu| < 0 @f$.
395  */
396  template<typename _Tp>
397  struct __gamma_temme_t
398  {
399  /// The input parameter of the gamma functions
400  _Tp __mu_arg;
401 
402  /// The output function @f$ 1/\Gamma(1 + \mu) @f$
403  _Tp __gamma_plus_value;
404 
405  /// The output function @f$ 1/\Gamma(1 - \mu) @f$
406  _Tp __gamma_minus_value;
407 
408  /// The output function @f$ \Gamma_1(\mu) @f$
409  _Tp __gamma_1_value;
410 
411  /// The output function @f$ \Gamma_2(\mu) @f$
412  _Tp __gamma_2_value;
413  };
