61  /**
62  * Return the reperiodized sine of argument x:
63  * @f[
64  * \sin_\pi(x) = \sin(\pi x)
65  * @f]
66  */
67  template<typename _Tp>
68  _Tp
69  __sin_pi(_Tp __x)
70  {
71  constexpr _Tp _S_pi = __gnu_cxx::__math_constants<_Tp>::__pi;
72  if (std::isnan(__x))
73  return std::numeric_limits<_Tp>::quiet_NaN();
74  else if (__x < _Tp{0})
75  return -__sin_pi(-__x);
76  else if (__x < _Tp{0.5L})
77  return std::sin(__x * _S_pi);
78  else if (__x < _Tp{1})
79  return std::sin((_Tp{1} - __x) * _S_pi);
80  else
81  {
82  auto __nu = std::floor(__x);
83  auto __arg = __x - __nu;
84  auto __sign = (int(__nu) & 1) == 1 ? -1 : +1;
85  auto __sinval = (__arg < _Tp{0.5L})
86  ? __sin_pi(__arg)
87  : __sin_pi(_Tp{1} - __arg);
88  return __sign * __sinval;
89  }
90  }

358  /**
359  * Reperiodized sincos.
360  */
361  template<typename _Tp>
362  __gnu_cxx::__sincos_t<_Tp>
363  __sincos_pi(_Tp __x)
364  {
365  constexpr auto _S_pi = __gnu_cxx::__math_constants<_Tp>::__pi;
366  constexpr auto _S_NaN = __gnu_cxx::__math_constants<_Tp>::__NaN;
367  if (std::isnan(__x))
368  return __gnu_cxx::__sincos_t<_Tp>{_S_NaN, _S_NaN};
369  else if (__x < _Tp{0})
370  {
371  __gnu_cxx::__sincos_t<_Tp> __tempsc = __sincos_pi(-__x);
372  return __gnu_cxx::__sincos_t<_Tp>{-__tempsc.sin_value,
373  __tempsc.cos_value};
374  }
375  else if (__x < _Tp{0.5L})
376  return __sincos(_S_pi * __x);
377  else if (__x < _Tp{1})
378  {
379  __gnu_cxx::__sincos_t<_Tp>
380  __tempsc = __sincos(_S_pi * (_Tp{1} - __x));
381  return __gnu_cxx::__sincos_t<_Tp>{__tempsc.sin_value,
382  -__tempsc.cos_value};
383  }
384  else
385  {
386  auto __nu = std::floor(__x);
387  auto __arg = __x - __nu;
388  auto __sign = (int(__nu) & 1) == 1 ? _Tp{-1} : _Tp{+1};
389 
390  auto __sinval = (__arg < _Tp{0.5L})
391  ? std::sin(_S_pi * __arg)
392  : std::sin(_S_pi * (_Tp{1} - __arg));
393  auto __cosval = std::cos(_S_pi * __arg);
394  return __gnu_cxx::__sincos_t<_Tp>{__sign * __sinval,
395  __sign * __cosval};
396  }
