40  /**
41  * @brief Return the sinus cardinal function
42  * @f[
43  * sinc(x) = \frac{\sin(x)}{x}
44  * @f]
45  */
46  template<typename Tp>
47  emsr::fp_promote_t<Tp>
48  sinc(Tp x)
49  {
50  if (std::isnan(x))
51  return emsr::quiet_NaN(x);
52  else if (std::abs(x) == emsr::infinity(x))
53  return Tp{0};
54  else if (std::abs(x) < emsr::sqrt_min(x))
55  return Tp{1} - x * x / Tp{6};
56  else
57  return std::sin(x) / x;
58  }

60  /**
61  * @brief Return the reperiodized sinus cardinal function
62  * @f[
63  * sinc_\pi(x) = \frac{\sin(\pi x)}{\pi x}
64  * @f]
65  */
66  template<typename Tp>
67  emsr::fp_promote_t<Tp>
68  sinc_pi(Tp x)
69  {
70  const auto s_pi = emsr::pi_v<Tp>;
71  if (std::isnan(x))
72  return emsr::quiet_NaN(x);
73  else if (std::abs(x) == emsr::infinity(x))
74  return Tp{0};
75  else
76  {
77  auto arg = s_pi * x;
78  if (std::abs(arg) < Tp{4} * emsr::sqrt_min(x))
79  return Tp{1} - arg * arg / Tp{6};
80  else
81  return emsr::sin_pi(x) / arg;
82  }
83  }

85  /**
86  * @brief Return the hyperbolic sinus cardinal function
87  * @f[
88  * sinhc(x) = \frac{\sinh(x)}{x}
89  * @f]
90  */
91  template<typename Tp>
92  emsr::fp_promote_t<Tp>
93  sinhc(Tp x)
94  {
95  if (std::isnan(x))
96  return emsr::quiet_NaN(x);
97  else if (std::abs(x) < Tp{4} * emsr::sqrt_min(x))
98  return Tp{1} + x * x / Tp{6};
99  else
100  return std::sinh(x) / x;
101  }

103  /**
104  * @brief Return the reperiodized hyperbolic sinus cardinal function
105  * @f[
106  * sinhc_\pi(x) = \frac{\sinh(\pi x)}{\pi x}
107  * @f]
108  */
109  template<typename Tp>
